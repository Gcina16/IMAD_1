package com.example.socialspark

import android.os.Bundle
import android.util.Log
import android.widget.Button
import android.widget.EditText
import android.widget.TextView
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        val etTime = findViewById<EditText>(R.id.etTimeOfDay)
        val btnSuggest = findViewById<Button>(R.id.btnSuggest)
        val btnReset = findViewById<Button>(R.id.btnReset)
        val tvSuggestion = findViewById<TextView>(R.id.tvSuggestion)

        btnSuggest.setOnClickListener {
		
            val input = etTime.text.toString().trim().lowercase()
            Log.d("SocialSpark", "User entered: $input")

            if (input.isEmpty()) {
                tvSuggestion.text = "Please enter a time of day."
                Log.e("SocialSpark", "Input is empty")
            } 
            else if (input == "morning") {
                tvSuggestion.text = "Send a 'Good morning' message to a family member."
            } 
            else if (input == "mid-morning") {
                tvSuggestion.text = "Reach out to a colleague with a quick 'Thank you.'"
            } 
            else if (input == "afternoon") {
                tvSuggestion.text = "Share a funny meme or interesting link with a friend."
            } 

            else if (input == "afternoon snack time") {
                tvSuggestion.text = "Send a quick 'Thinking of you' message."
            } 
            else if (input == "dinner") {
                tvSuggestion.text = "Call a friend or relative for a 5-minute catch-up."
            } 
            else if (input == "after dinner" || input == "night") {
                tvSuggestion.text = "Leave a thoughtful comment on a friend's post."
            }
            else {

                tvSuggestion.text = "Invalid time entered. Try Morning, Afternoon, or Dinner."
                Log.w("SocialSpark", "Invalid input: $input")
            }
        }

        btnReset.setOnClickListener {
            etTime.text.clear()
            tvSuggestion.text = ""
            Log.d("SocialSpark", "Input reset")
        }
    }
}
